The problem

According to news websites whose names I won’t mention ahem ahem rhymes with shmulf bews ahem ahem, I should expect rain almost once a week! I should be expecting it in the middle of Abu Dhabi July and packing a jacket just in case! I should be cancelling outdoor dates and carrying my umbrella everywhere I go… yeah sure.

SO, let’s learn how to (hopefully ethically) scrape their news articles for mentions of rain over the current decade and come up with a percentage likelihood for how much rain we’re expected to see according to each article that mentions rain.

The sitemap for gulfnews.com looks like this: https://gulfnews.com/sitemap/sitemap-daily-2026-02-21.xml

Each day, an entry. I just need to extract the contents of each day’s headlines and filter for keywords like ‘rain’, ‘rainfall’, ‘precipitation’, etc. and then save articles that match into a jsonl file sorted by month.

After python-ing with gpt5 on opencode using xmls and beautifulsoup4, I get something that looks like this:

For each url found in the day’s sitemap, we skim the title to see if anything relates to weather. If it does, go to the url and extract the article body to form this jsonl file. The format looks like this:

The main function in charge of doing this is def extract_sitemap_urls_to_jsonl() in ../main.py

Code 
import sys
from pathlib import Path

sys.path.append(str(Path(__file__).resolve().parent.parent))

from main import *
def extract_sitemap_urls_to_jsonl(
    start_date,
    end_date,
    output_root=OUTPUT_ROOT,
    request_delay_s=0.5,
    force=False,
):
    output_path = None
    header = (
        f"{ANSI_INFO}╔══════════════════════════════════════════════╗{ANSI_RESET}\n"
        f"{ANSI_INFO}║{ANSI_RESET} {ANSI_BOLD}Gulf News Extraction Console{ANSI_RESET}           {ANSI_INFO}║{ANSI_RESET}\n"
        f"{ANSI_INFO}╚══════════════════════════════════════════════╝{ANSI_RESET}"
    )
    print(header, flush=True)

    # We'll generate a sitemap url based on the format of the website:
    #    "https://gulfnews.com/sitemap/sitemap-daily-2026-02-21.xml"
    prefix = "https://gulfnews.com/sitemap/sitemap-daily"

    current_date = parse_date(start_date)
    end = parse_date(end_date)

    while current_date < end:
        sitemap_url = f"{prefix}-{current_date.isoformat()}.xml"
        sitemap_date = current_date.isoformat()
        output_path = output_path_for_sitemap(sitemap_date, output_root=output_root)
        # This prevents us from re-bothering the server for data we've already
        # collected, unless the force option is passed.
        if output_path.exists() and not force:
            return output_path

        xml_text = fetch_text(sitemap_url)
        entries = parse_sitemap(xml_text)
        total = len(entries)
        counter = 0
        print(
            f"{ANSI_INFO}[INFO]{ANSI_RESET} "
            f"Date: {ANSI_BOLD}{sitemap_date}{ANSI_RESET} | "
            f"Entries: {ANSI_BOLD}{total}{ANSI_RESET}",
            flush=True,
        )

        # Make a different JSONL for each month of entries that've been filtered
        # through.
        output_path.parent.mkdir(parents=True, exist_ok=True)

        matches = 0
        with output_path.open("w", encoding="utf-8") as handle:
            for entry in entries:
                counter += 1
                progress = f"{counter}/{total}" if total else "0/0"
                print(
                    f"\r{ANSI_FETCH}[FETCH]{ANSI_RESET} "
                    f"{ANSI_BOLD}{progress}{ANSI_RESET} "
                    f"{ANSI_DIM}{current_date}{ANSI_RESET}",
                    end="",
                    flush=True,
                )
                html_text = fetch_text(entry.url)
                soup = BeautifulSoup(html_text, "html.parser")
                title = extract_title(soup)
                if title == "":
                    print(f"\n{ANSI_WARN}[WARN]{ANSI_RESET} Empty title", flush=True)
                matched = title_matches_keywords(title)
                if matched:
                    matches += 1
                    article_text = extract_article_text(soup)
                else:
                    article_text = None
                handle.write(
                    json.dumps(
                        {
                            "url": entry.url,
                            "lastmod": (
                                entry.lastmod.isoformat() if entry.lastmod else None
                            ),
                            "title": title,
                            "title_matches_keywords": matched,
                            "text": article_text,
                        }
                    )
                    + "\n"
                )
                if request_delay_s:
                    sleep(request_delay_s)
            print(
                f"\r{ANSI_DONE}[DONE]{ANSI_RESET} Day complete with #{matches=}",
                flush=True,
            )

        current_date += timedelta(days=1)

    return output_path

This process is very slow since we’re fetching each url from each day one at a time, we can parallelize this process by using threads or asyncio but we shouldn’t do that wantonly. Hitting a news site with dozens of requests per second is not very polite and respectful. So instead, we’re going to parallelize ethically by limiting our overall HTTP calls.

Some Statistics

Let’s start with a simple frequency count on how many times rain has been mentioned over this decade. To do this, we’ll first filter out all the JSONL logs to only those that have their title_matches_keywords == true.

How do we filter? Let’s make a script to take all the jsonl logs and filter them out to leave only the ones with our keywords.

Code 
def iter_jsonl_lines(path):
    with path.open("r", encoding="utf-8") as handle:
        for lineno, line in enumerate(handle, start=1):
            line = line.strip()
            if line:
                yield lineno, line

def filter_jsonl_file(src, dst, field):
    dst.parent.mkdir(parents=True, exist_ok=True)

    kept = 0
    total = 0
    bad = 0

    tmp = dst.with_suffix(dst.suffix + ".tmp")

    with tmp.open("w", encoding="utf-8") as out:
        for lineno, line in iter_jsonl_lines(src):
            total += 1
            try:
                obj = json.loads(line)
            except json.JSONDecodeError:
                bad += 1
                continue

            if bool(obj.get(field)) is True:
                out.write(line)
                out.write("\n")
                kept += 1

    tmp.replace(dst)
    return kept, total, bad


from pathlib import Path
def filter_jsonls(
    src_root = Path("data/gulfnews/articles"),
    out_root = Path("data/gulfnews/articles.filtered"),
    field= "title_matches_keywords",
    ):
    total_files = 0
    total_lines = 0
    total_kept = 0
    total_bad = 0

    files = sorted(src_root.glob("**/*.jsonl"))

    for src in files:
        if not src.is_file():
            continue
        rel = src.relative_to(src_root)
        dst = out_root / rel
        kept, total, bad = filter_jsonl_file(src, dst, field)

        total_files += 1
        total_lines += total
        total_kept += kept
        total_bad += bad

    print(
        f"Processed {total_files} files | "
        f"lines: {total_lines} | kept: {total_kept} | bad_json: {total_bad} | "
        f"out: {out_root}"
    )

from datetime import date, datetime, timedelta


def parse_date(value):
    if value is None:
        return None
    return date.fromisoformat(value)


def date_from_filename(path):
    name = path.name
    date_part = name.replace("sitemap-", "").replace(".jsonl", "")
    return date.fromisoformat(date_part)


def count_jsonl_lines(path):
    count = 0
    with path.open("r", encoding="utf-8") as handle:
        for line in handle:
            if line.strip():
                count += 1
    return count


def daterange(start, end):
    current = start
    while current <= end:
        yield current
        current += timedelta(days=1)


def rolling_mean(values, window):
    if window <= 1:
        return list(values)
    result = []
    running = 0
    for idx, value in enumerate(values):
        running += value
        if idx >= window:
            running -= values[idx - window]
            result.append(running / window)
        else:
            result.append(running / (idx + 1))
    return result


def aggregate_counts(dates, counts, window_days):
    if window_days <= 0:
        raise SystemExit("aggregate_days must be positive")
    buckets = []
    bucket_counts = []
    bucket_start = dates[0]
    running = 0
    for day, count in zip(dates, counts):
        if (day - bucket_start).days >= window_days:
            buckets.append(bucket_start)
            bucket_counts.append(running)
            bucket_start = day
            running = 0
        running += count
    buckets.append(bucket_start)
    bucket_counts.append(running)
    return buckets, bucket_counts


def plot_counts_of_rain():
    aggregate_days = 7
    rolling_window = 4  # in number of bars
    root_filt = Path(root_filt)
    root_unfilt = Path(root_unfilt)
    output_path = Path(out)

    def get_count_for_dates(root):
        jsonl_files = sorted(root.glob("**/sitemap-*.jsonl"))
        if not jsonl_files:
            raise SystemExit(f"No JSONL files found under {root}")

        daily_counts = {}
        for path in jsonl_files:
            day = date_from_filename(path)
            daily_counts[day] = count_jsonl_lines(path)

        return daily_counts

    daily_filtered = get_count_for_dates(root_filt)
    daily_unfiltered = get_count_for_dates(root_unfilt)

    start = min(min(daily_filtered), min(daily_unfiltered))
    end = max(max(daily_filtered), max(daily_unfiltered))
    if start > end:
        raise SystemExit("Start date must be before end date")

    dates = list(daterange(start, end))
    counts = [daily_filt.get(day, 0) for day in dates]
    counts_unfilt = [daily_unfilt.get(day, 0) for day in dates]

    bucket_dates, counts = aggregate_counts(dates, counts, aggregate_days)
    _, counts_unfilt = aggregate_counts(dates, counts_unfilt, aggregate_days)

    smoothed = rolling_mean(counts, window=rolling_window)
    smoothed_unfilt = rolling_mean(counts_unfilt, window=rolling_window)

    unfiltered_scale = 15
    counts_unfilt_scaled = [value / unfiltered_scale for value in counts_unfilt]
    counts_filtered_neg = [-value for value in counts]
    smoothed_filtered_neg = [-value for value in smoothed]
    smoothed_unfilt_scaled = [value / unfiltered_scale for value in smoothed_unfilt]

    fig, ax = plt.subplots(figsize=(11, 7.5), dpi=180)
    ax.set_facecolor("#EAEAF2")  # The seaborn background...
    ax.axvline(0, color="#333333", linewidth=1.0, alpha=0.6)
    ax.grid(True, axis="x", alpha=0.25)

    y_positions = list(range(len(bucket_dates)))
    ax.barh(
        y_positions,
        counts_filtered_neg,
        color="#1f77b4",
        alpha=0.75,
        height=0.9,
        label="Filtered",
    )
    ax.barh(
        y_positions,
        counts_unfilt_scaled,
        color="#d62728",
        alpha=0.75,
        height=0.9,
        label=f"Unfiltered (scaled /{unfiltered_scale})",
    )
    ax.plot(
        smoothed_filtered_neg,
        y_positions,
        color="#6400ff",
        linewidth=1.6,
        label=f"Filtered mean ({rolling_window}-bar)",
    )
    ax.plot(
        smoothed_unfilt_scaled,
        y_positions,
        color="#ff7f0e",
        linewidth=1.6,
        label=f"Unfiltered mean ({rolling_window}-bar)",
    )

    ax.set_title("News Articles per Period (Filtered vs Unfiltered)")
    ax.set_xlabel(
        f"Articles per {aggregate_days} days (filtered left, unfiltered right)"
    )
    ax.set_ylabel("Period start")
    ax.legend()

    def format_tick(value, _):
        value = abs(value)
        if value.is_integer():
            return f"{int(value)}"
        return f"{value:.1f}"

    ax.xaxis.set_major_formatter(mticker.FuncFormatter(format_tick))

    step = max(1, len(bucket_dates) // 12)
    ax.set_yticks(y_positions[::step])
    ax.set_yticklabels([day.isoformat() for day in bucket_dates[::step]])
    ax.invert_yaxis()

    fig.tight_layout()
    output_path.parent.mkdir(parents=True, exist_ok=True)
    fig.savefig(output_path)
    plt.show()

    total = sum(counts)
    peak = max(counts) if counts else 0
    peak_date = bucket_dates[counts.index(peak)] if counts else None
    print(f"Saved plot to {output_path}")
    print(f"Total articles: {total}")
    if peak_date:
        print(f"Peak period start: {peak_date} ({peak})")

I just need to extract the contents of each day's headlines and filter for
keywords like 'rain', 'rainfall', 'precipitation', etc. and then save articles

import sys
from pathlib import Path

sys.path.append(str(Path(__file__).resolve().parent.parent))

from main import *
def extract_sitemap_urls_to_jsonl(
    start_date,
    end_date,
    output_root=OUTPUT_ROOT,
    request_delay_s=0.5,
    force=False,
):
    output_path = None
    header = (
        f"{ANSI_INFO}╔══════════════════════════════════════════════╗{ANSI_RESET}\n"
        f"{ANSI_INFO}║{ANSI_RESET} {ANSI_BOLD}Gulf News Extraction Console{ANSI_RESET}           {ANSI_INFO}║{ANSI_RESET}\n"
        f"{ANSI_INFO}╚══════════════════════════════════════════════╝{ANSI_RESET}"
    )
    print(header, flush=True)

    # We'll generate a sitemap url based on the format of the website:
    #    "https://gulfnews.com/sitemap/sitemap-daily-2026-02-21.xml"
    prefix = "https://gulfnews.com/sitemap/sitemap-daily"

    current_date = parse_date(start_date)
    end = parse_date(end_date)

    while current_date < end:
        sitemap_url = f"{prefix}-{current_date.isoformat()}.xml"
        sitemap_date = current_date.isoformat()
        output_path = output_path_for_sitemap(sitemap_date, output_root=output_root)
        # This prevents us from re-bothering the server for data we've already
        # collected, unless the force option is passed.
        if output_path.exists() and not force:
            return output_path

        xml_text = fetch_text(sitemap_url)
        entries = parse_sitemap(xml_text)
        total = len(entries)
        counter = 0
        print(
            f"{ANSI_INFO}[INFO]{ANSI_RESET} "
            f"Date: {ANSI_BOLD}{sitemap_date}{ANSI_RESET} | "
            f"Entries: {ANSI_BOLD}{total}{ANSI_RESET}",
            flush=True,
        )

        # Make a different JSONL for each month of entries that've been filtered
        # through.
        output_path.parent.mkdir(parents=True, exist_ok=True)

        matches = 0
        with output_path.open("w", encoding="utf-8") as handle:
            for entry in entries:
                counter += 1
                progress = f"{counter}/{total}" if total else "0/0"
                print(
                    f"\r{ANSI_FETCH}[FETCH]{ANSI_RESET} "
                    f"{ANSI_BOLD}{progress}{ANSI_RESET} "
                    f"{ANSI_DIM}{current_date}{ANSI_RESET}",
                    end="",
                    flush=True,
                )
                html_text = fetch_text(entry.url)
                soup = BeautifulSoup(html_text, "html.parser")
                title = extract_title(soup)
                if title == "":
                    print(f"\n{ANSI_WARN}[WARN]{ANSI_RESET} Empty title", flush=True)
                matched = title_matches_keywords(title)
                if matched:
                    matches += 1
                    article_text = extract_article_text(soup)
                else:
                    article_text = None
                handle.write(
                    json.dumps(
                        {
                            "url": entry.url,
                            "lastmod": (
                                entry.lastmod.isoformat() if entry.lastmod else None
                            ),
                            "title": title,
                            "title_matches_keywords": matched,
                            "text": article_text,
                        }
                    )
                    + "\n"
                )
                if request_delay_s:
                    sleep(request_delay_s)
            print(
                f"\r{ANSI_DONE}[DONE]{ANSI_RESET} Day complete with #{matches=}",
                flush=True,
            )

        current_date += timedelta(days=1)

    return output_path

def iter_jsonl_lines(path):
    with path.open("r", encoding="utf-8") as handle:
        for lineno, line in enumerate(handle, start=1):
            line = line.strip()
            if line:
                yield lineno, line

def filter_jsonl_file(src, dst, field):
    dst.parent.mkdir(parents=True, exist_ok=True)

    kept = 0
    total = 0
    bad = 0

    tmp = dst.with_suffix(dst.suffix + ".tmp")

    with tmp.open("w", encoding="utf-8") as out:
        for lineno, line in iter_jsonl_lines(src):
            total += 1
            try:
                obj = json.loads(line)
            except json.JSONDecodeError:
                bad += 1
                continue

            if bool(obj.get(field)) is True:
                out.write(line)
                out.write("\n")
                kept += 1

    tmp.replace(dst)
    return kept, total, bad

            continue
        rel = src.relative_to(src_root)
        dst = out_root / rel
        kept, total, bad = filter_jsonl_file(src, dst, field)

        total_lines += total
        total_kept += kept
        total_bad += bad

from datetime import date, datetime, timedelta


def parse_date(value):
    if value is None:
        return None
    return date.fromisoformat(value)


def date_from_filename(path):
    name = path.name
    date_part = name.replace("sitemap-", "").replace(".jsonl", "")
    return date.fromisoformat(date_part)


def count_jsonl_lines(path):
    count = 0
    with path.open("r", encoding="utf-8") as handle:
        for line in handle:
            if line.strip():
                count += 1
    return count


def daterange(start, end):
    current = start
    while current <= end:
        yield current
        current += timedelta(days=1)


def rolling_mean(values, window):
    if window <= 1:
        return list(values)
    result = []
    running = 0
    for idx, value in enumerate(values):
        running += value
        if idx >= window:
            running -= values[idx - window]
            result.append(running / window)
        else:
            result.append(running / (idx + 1))
    return result


def aggregate_counts(dates, counts, window_days):
    if window_days <= 0:
        raise SystemExit("aggregate_days must be positive")
    buckets = []
    bucket_counts = []
    bucket_start = dates[0]
    running = 0
    for day, count in zip(dates, counts):
        if (day - bucket_start).days >= window_days:
            buckets.append(bucket_start)
            bucket_counts.append(running)
            bucket_start = day
            running = 0
        running += count
    buckets.append(bucket_start)
    bucket_counts.append(running)
    return buckets, bucket_counts
